# 철벽 보안 알고리즘 [[link]](https://www.acmicpc.net/problem/9322)
**사용언어: C++14**

![캡처](https://user-images.githubusercontent.com/38516906/60842938-4dcd7e00-a210-11e9-8ee9-7b0189932321.PNG)

```
#include <iostream>
#include <string>

using namespace std;
 
int main() 
{
    int tc;
    cin >> tc;
    while (tc--) {
        int n, idx[1001];
        string word[3][1001];
        cin >> n;
        for (int j = 0; j < 3; j++) {
            for (int i = 0; i < n; i++) { cin >> word[j][i]; }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (word[0][i] == word[1][j]) {
                    idx[j] = i;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (idx[j] == i){ cout << word[2][j] << " "; }
            }
        }
        cout << '\n';
    }
    return 0;
}
```
